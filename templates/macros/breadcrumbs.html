{#- Breadcrumb macros. Usage from a page template: {% from "macros/breadcrumbs.html" import breadcrumb, breadcrumb_jsonld %} {{ breadcrumb([ ('/talks/', 'Talks'), (none, this.title), ]) }} {{ breadcrumb_jsonld([ ('/talks/', 'Talks'), (this|url, this.title), ]) }} Each entry is a (path, label) tuple. The first crumb (Home) is added automatically. Use `none` as the path for the current page so it doesn't render as a link in the visible breadcrumb. The JSON-LD variant requires every entry to have a path (use `this|url` for the current page). -#} {%- macro breadcrumb(trail) -%} {%- set branding = bag('branding') -%}
  1. Home
    2. {%- for path, label in trail -%}
  2. {%- if path and not loop.last -%} {{ label }} {%- else -%} {{ label }} {%- endif -%}
    3. {%- endfor -%}
{%- endmacro -%} {%- macro breadcrumb_jsonld(trail) -%} {%- set host = bag('branding')['canonical_host'] -%} {%- endmacro -%}
