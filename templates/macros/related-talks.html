{#- Related-talks panel rendered at the bottom of a talk page. Sections (rendered only if non-empty): 1. More by this speaker — other talks by any of this talk's speakers (same edition first, then archive) 2. Same track — other talks in the same edition with the same track 3. Also in this edition — sibling talks (capped, deterministic order) -#} {%- macro related_panel(talk, my_speakers, talks_root, speakers_root, archive_year) -%} {%- set my_code = talk.code or talk._id -%} {#- 1. More by this speaker -#} {%- set by_speaker = [] -%} {%- if my_speakers -%} {%- for sp in my_speakers -%} {%- for code in (sp.talks or []) -%} {%- if code != my_code -%} {%- set sibling = site.get(talks_root ~ '/' ~ code) -%} {%- if sibling and (by_speaker|length) < 8 -%} {%- if sibling not in by_speaker -%} {%- set _ = by_speaker.append(sibling) -%} {%- endif -%} {%- endif -%} {%- endif -%} {%- endfor -%} {%- endfor -%} {%- endif -%} {#- 2. Same track this edition -#} {%- set same_track = [] -%} {%- set talks_index = site.get(talks_root) -%} {%- if talk.track and talks_index -%} {%- for sibling in talks_index.children -%} {%- if sibling.track == talk.track and sibling.code != my_code -%} {%- if (same_track|length) < 6 and sibling not in by_speaker -%} {%- set _ = same_track.append(sibling) -%} {%- endif -%} {%- endif -%} {%- endfor -%} {%- endif -%} {#- 3. Also in this edition (deterministic: alpha by title, take 4) -#} {%- set seen = (by_speaker + same_track) -%} {%- set also_in = [] -%} {%- if talks_index -%} {%- for sibling in talks_index.children|sort(attribute='title') -%} {%- if sibling.code != my_code and sibling not in seen and (also_in|length) < 4 -%} {%- set _ = also_in.append(sibling) -%} {%- endif -%} {%- endfor -%} {%- endif -%} {%- if by_speaker or same_track or also_in -%}

{%- endif %} {%- if same_track %}

    {%- for t in same_track %}

    • {%- endfor %}
{%- endif %} {%- if also_in %}

    {%- for t in also_in %}
  • {{ t.title }} {%- if t.track %}{{ t.track }}{% endif -%}
    • {%- endfor %}
{%- endif %}
{%- endif %} {%- endmacro -%}
